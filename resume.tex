% Defines a single resume item.
%     Parameters:
%         #1 => Job Title 
%         #2 => Company
%         #3 => Time Period OPTIONALLY EMPTY
%         #4 => Description items (CDL of strings)
%         #5 => List of tools used OPTIONALLY EMPTY
\newcommand{\resumeitem}[5]{
    \subsection{#1 --- #2 \ifstrempty{#3}{} {\hfill #3}}
        \begin{itemize}[nosep]
            \foreach\x in #4 { \item \x }
        \end{itemize}
        \ifstrempty{#5}{}{
            \subsubsection{Tools} #5
        }
}
